pipeline {
	agent any
	    stages {
		   stage('Build Docker image') {
			steps {
			script {
   				docker.build('fibonacci:3.0', '.')
			}
			}
		   }
		   stage('Push to Docker Hub') {
			steps {
				script{
				docker.withRegistry('https://hub.docker.com', 'Docker_Login') {
                        		def dockerImage = docker.image('fibonacci:3.0')
					dockerImage.push('shreyam2001/fibonacci-2:3.0')
                    		}
				}
			}
		   }
	     }
}
