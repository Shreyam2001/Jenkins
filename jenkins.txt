	agent any
	    stages {

   				def dockerImage = docker.build('shreyam2001/fibonacci-2/fibonacci:3.0', '.')
			}

                        		docker.image('shreyam2001/fibonacci-2/fibonacci:3.0').push()
                    		}
				}
			}
		   }
	     }
}
