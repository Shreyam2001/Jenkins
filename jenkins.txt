pipeline {
	agent any
	    stages {
		   stage('Build Docker image') {
			steps {
			script {
   				docker.build('fibonacci:3.0', '.')
			}
			}
		   }
		   stage('Push to Docker Hub') {
			steps {
				script{
				docker.withRegistry('https://index.docker.io/v1/', 'Docker_Login') {
					docker.image('fibonacci:3.0').push('fibonacci-2/fibonacci:3.0')
                    		}
				}
			}
		   }
	     }
}
